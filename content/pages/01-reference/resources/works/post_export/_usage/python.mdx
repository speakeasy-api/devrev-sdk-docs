),
    issue=sdk.WorksFilterIssue(

    ],

),
    ticket=sdk.WorksFilterTicket(

            'string',

        ],
        sla_summary=sdk.SLASummaryFilter(),
        source_channel=[
            'string',
        ],

    ],
)

res = s.works.post_export(req)
